	G1= ({S}, {a,b}, P, S)
	P :
	S → aSb
	S → ab
	S ⇒ ab (menor cadeia)
	S ⇒ aSb ⇒ aabb
	S ⇒ aSb ⇒ aaSbb ⇒ aaabbb

	G2 = ({S}, {a,b}, P, S)
	P :
	S → aAb
	A → ab|λ|S
	S ⇒ aAb ⇒ ab (menor cadeia)
	S ⇒ aAb ⇒ aabb
	S ⇒ aAb ⇒ aSb ⇒ aaAbb ⇒ aaabbb
